55 def find_nearest_blind_tones(kidslist):
56  """ KID信号の左右にある最も近いblind信号を探す
57  引数
58  ----
59  kidslist(dmerge.read_kidslist()関数の戻り値)
60 
61  戻り値
62  ------
63  list [[KID信号, 左blind tone, 右blind tone], ...]
64 
65  戻り値の例
66  ----------
67  [[-196483 -201203 -192949]
68  [-189634 -192949 -184695]
69  [-188948 -192949 -184695]
70  ...
71  [ 154774 153170 161962]
72  [ 158692 153170 161962]
73  [ 169653 161962 171230]]
74 
75  詳細
76  ----
77  この関数は引数に与えられたkidslistに含まれるkidsとblindsを利用する。
78  kidsとblindsを組み合わせて各kid信号の両側にある最も近いblind信号を探す。
79  見つかった左右のblind信号とkid信号をリストにまとめて返す。戻り値はリストの入れ子構造となる。
80  blind信号が見つからない場合はNoneが格納される。
81  戻り値はkidのbin番号で昇順になっている。
82  この関数はbbsweeplib.TODcalibに定義されていたkids_with_both_blinds()関数を単機能にしたもの。
83 
84  探索の方法
85  ----------
86  kid信号とblind信号を合わせたリストを作成する。このリストはbin番号で昇順に並べ替えられている。
87  あるkid信号のindexを取得し、そこを起点に左右に探索する(indexを減少または増加させる)。起点は含まない。
88  見つかった信号のbin番号がblind信号のリストに含まれる場合は、それをそのkid信号のblind信号とする。
89  """
90  info, kids, blinds, powers = kidslist
91  all_tones = sorted(list(kids) + list(blinds))
92  result = []
93  for kid in sorted(kids):
94  left_tone = None
95  i = all_tones.index(kid) - 1
96  while i >= 0:
97  tone = all_tones[i]
98  if tone in blinds:
99  left_tone = tone
100  break
101  i -= 1
102  right_tone = None
103  i = all_tones.index(kid) + 1
104  while i < len(all_tones):
105  tone = all_tones[i]
106  if tone in blinds:
107  right_tone = tone
108  break
109  i += 1
110  result.append([kid, left_tone, right_tone])
111  return result

323 def find_peaks(freq, ampl, fc, smooth=None, threshold=None, minq=10000, maxratio=0.5):
324  """Search peaks in data.
325  引数
326  ----
327  freq 周波数の配列
328  ampl amplitudeの配列
329  fc TODのキャリア周波数
330  smooth 平滑化のwindow長(正の整数)[オプション]
331  threshold ピークの判定基準[オプション]
332  minq Qの最小値(ピークの良し悪しの判定に利用する)[オプション]
333  maxratio (bg - d)/bgの最小値(ピークの良し悪しの判定に利用する)[オプション]
334 
335  戻り値
336  ------
337  list Lorentzianのフィット共にgaolinbgのパラメタの初期値を決定するのに必要なパラメタを辞書にまとめたリスト
338  """
339  dfreq = freq[1] - freq[0] # frequency resolution
340  if smooth is None:
341  smooth = 15
342 
343  # 平滑化
344  deriv2 = scipy.signal.savgol_filter(ampl, oddify(smooth), 3, 2, delta=dfreq)
345  ampl_s = scipy.signal.savgol_filter(ampl, oddify(smooth), 3, 0, delta=dfreq)
346 
347  # ピーク探索
348  if threshold is None:
349  d2sigma = np.std(deriv2)
350  d2sigma = np.std(deriv2[np.abs(deriv2)<2*d2sigma])
351  threshold = 3*d2sigma
352  kid_indices = []
353  for i in np.where(deriv2 > threshold)[0]:
354  if i < len(deriv2) - 1 and deriv2[i - 1] <= deriv2[i] and deriv2[i] >= deriv2[i + 1]:
355  kid_indices.append(i)
356 
357  kids = [] # 各KIDをLorentzianでフィットした結果(Lorentzianのパラメタ)
358  nbadq = 0 # Qが条件を満たさなかったKIDの数
359  nbadd = 0 # maxratioを満たさなかったKIDの数
360  if not kid_indices:
361  return []
362 
363  for i in kid_indices:
364  #
365  # 詳細
366  # ----
367  # ピークの左右でamplitudeの2階の微係数が0になる部分(変曲点)を利用してフィット範囲を設定する。
368  # フィットの結果Qと(bg-d)/bgが条件を満たさない場合は、そのピークは無視される。
369  #
370  l, r = i, i
371  while l > 0 and deriv2[l] > 0: l -= 1
372  while r < len(deriv2) and deriv2[r] > 0: r += 1
373  w = (r - l + 1)*dfreq
374  w = w*6.0/np.sqrt(3) # convert to FWHM
375  l = int((l - i)*6.0/np.sqrt(3) + i)
376  r = int((r - i)*6.0/np.sqrt(3) + i)
377  if l < 0 : l = 0
378  if r >= len(freq): r = len(freq) - 1
379 
380  q0 = freq[i]/w
381  f, q, d, bg = fitLorentzian(freq[l:r], ampl[l:r], freq[i], q0)
382 
383  if (bg - d)/bg > maxratio:
384  n = 10
385  if l - n >= 0 and r + n < len(freq):
386  f, q, d, bg = fitLorentzian(freq[l-n:r+n], ampl[l-n:r+n], freq[i], q0)
387  print('rough refitting with Lorentzian...')
388  if q < minq:
389  nbadq += 1
390  continue
391  if (bg - d)/bg > maxratio:
392  nbadd += 1
393  continue
394  kids.append((f, q, d, bg))
395  del l, r, f, q, d, bg
396  if nbadq > 0:
397  print('removed', nbadq, 'peaks with bad Q')
398  if nbadd > 0:
399  print('removed', nbadd, 'peaks with bad S21min')
400  kids.sort() #sort by frequency
401 
402  # pick up a peak which is closest to the carrier frequency (when fc in freq range)
403  if len(kids) > 0 and fc >= freq[0] and fc <= freq[-1]:
404  i = np.argmin(abs(np.array(kids).T[0] - fc))
405  kids = [kids[i]]
406 
407  # gaolinbgのパラメタの初期値を決めるためのbackgroundなどを計算する
408  for i, (f, q, depth, bg) in enumerate(kids):
409  f0ind = np.argmin(abs(freq - f))
410  w = f/q
411  dl = w/2.0/dfreq
412  dr = w/2.0/dfreq
413  bg_l = ampl_s[max(int(f0ind - 3*dl), 0)]
414  if int(f0ind - 3*dl)<0:
415  bg_r = ampl_s[0]
416  else:
417  bg_r = ampl_s[min(int(f0ind - 3*dl), len(freq) - 1)]
418  a_off = (bg_l + bg_r)/2.0
419  a_on = ampl_s[f0ind]
420  kids[i] = {
421  'Q': q,
422  'f0': f,
423  'f0ind': f0ind,
424  'dl': int(dl),
425  'dr': int(dr),
426  'a_off': a_off,
427  'a_on': a_on,
428  'ampl': depth,
429  'bg': bg
430  }
431  return kids

445 def fitLorentzian(freq, ampl, f0, q0):
446  """Fit data with Lorentzian curve.
447  引数
448  ----
449  freq a 1-D array of frequency
450  ampl a 1-D array of amplitude
451  f0 initial parameter for center frequency
452  q0 initial parameter for quality factor
453 
454  戻り値
455  ------
456  (fc, q, d, bg) フィットの結果をまとめたタプル
457 
458  パラメタの説明
459  --------------
460  f : center frequency
461  q : quality factor
462  d : amplitude for Lorentzian curve
463  bg: constant background level
464  """
465  def f(x):
466  (a, b, c, d) = x # background, amplitude, 2/FWHM, freq. center
467  y = a + b / (((freq-d)*c)**2 + 1)
468  return y - ampl
469  a = np.median(ampl)
470  b = -0.8 * a
471  c = 2.0 * q0 / f0
472  d = f0
473  x0 = np.array([a, b, c, d])
474  x1 = scipy.optimize.leastsq(f, x0)[0]
475  (a, b, c, d) = x1
476  fc = d
477  q = abs(c * d / 2.0)
478  return (fc, q, -b, a)

480 def adjust_fitrange(nfwhm, ndata, nparams, peakparams, factor=1):
481  """Adjust fit range to make sure :math:`N_{free} \geq nparam`
482  引数
483  ----
484  float nfwhm: minimum fit length in unit of FWHM
485  integer ndata: length of data
486  integer nparams: number of parameter
487  dict peakparams: peak information
488  integer factor
489 
490  戻り値
491  ------
492  tuple (rbegin, rend) フィット範囲の始まりと終わりのインデックス
493 
494  factorについて
495  --------------
496  Specify 2 as 'factor' if you fit complex variable(as 2d-vector).
497 
498  利用するピークパラメタ
499  ----------------------
500  - f0ind: a integer, index of peak center
501  - dl : a integer, index count to reach the left half-maximum
502  - dr : a integer, index count to reach the right half-maximum
503  """
504  f0ind = peakparams['f0ind']
505  l, c, r = f0ind-peakparams['dl'], f0ind, f0ind+peakparams['dr']
506  if (r - l + 1)*factor >= nparams:
507  rbegin, rend = int(c-nfwhm*(c-l)), int(c+nfwhm*(r-c))
508  else:
509  n = nparams/(float(r - l + 1)*factor)
510  rbegin, rend = int(f0ind-nfwhm*n*(c-l)), int(c+nfwhm*n*(r-c))
511  if rbegin < 0:
512  if rend + (-rbegin) >= ndata:
513  rend = ndata-1
514  rbegin = 0
515  else:
516  rend = rend + (-rbegin)
517  rbegin = 0
518  if rend >= ndata:
519  if rbegin - (rend - ndata) < 0:
520  rbegin = 0
521  rend = ndata-1
522  else:
523  rbegin = rbegin - (rend - ndata)
524  rend = ndata-1
525  if (rend - rbegin + 1)*factor < 11:
526  raise Exception("Fit range guess error")
527  return slice(rbegin, rend)

587 def find_glitch(yss,
588  baseline_thresh = 6.0, glitch_thresh = 5.0, clusterize_thresh = 2, offset = 0,
589  opt_diff2=True, smrange=1, returnval=False):
590  """
591  find glitches common to `yss`, assuming glitch exists at the same
592  time of all data in `yss`. Too close glitches or broad glitch are
593  treated as one glitch.
594 
595  引数
596  ----
597  yss : an array of 1-D arrays of data to deglitch
598  baseline_thresh : threshold to decide baseline
599  glitch_thresh : threshold to decide as glitch
600  clusterize_thresh: if gap between glitches are less than or equal to this, treat them as one glitch.
601  offset : an integer offset added to returned bad array
602  opt_diff2 : bool to select whether to use 2nd derivative for glitch identification
603  smrange : an integer to set range of rolling mean (used for glitch identification based on std)
604  returnval : bool to choose whether to return threshold value or not
605 
606  戻り値(returnvalの値によって変化する)
607  -------------------------------------
608  an 1-D array of boolean, that can be used as index of ys (elements of yss).
609  (option) threshold value for glitch identification (return when returnval=True)
610  """
611  ave = np.average(yss, axis=0)
612  xs = np.arange(len(yss[0]))
613  dx = xs[1] - xs[0]
614  if opt_diff2:
615  diff2 = np.array( _numdif_2(ave, dx) )
616  sigma = np.std(diff2)
617  good = (np.abs(diff2) < (baseline_thresh*sigma))
618  sigma = np.std(diff2[good])
619  bad = (np.abs(diff2) >= (glitch_thresh*sigma))
620  if returnval:
621  thre = glitch_thresh*sigma/np.sqrt(6)*(dx**2)
622  else:
623 
624  weights = np.ones(smrange)/smrange
625  assert( len(ave)>len(weights) )
626  smooth = np.convolve(ave, weights, 'same')
627 
628  invalid = int( (len(weights)-1)/2 )
629  smooth[:invalid] = np.mean(ave[:invalid]) * np.ones( len(ave[:invalid]) )
630  smooth[-invalid-1:] = np.mean(ave[-invalid-1:]) * np.ones( len(ave[-invalid-1:]) )
631 
632  ave_sm = ave - smooth
633 
634  mean = np.mean(ave_sm)
635  sigma = np.std(ave_sm)
636  good = (np.abs(ave_sm-mean) < (baseline_thresh*sigma))
637  mean = np.mean(ave_sm[good])
638  sigma = np.std(ave_sm[good])
639  bad = (np.abs(ave_sm-mean) >= (glitch_thresh*sigma))
640  if returnval:
641  thre = glitch_thresh*sigma
642 
643  bad = _clusterize_indices(bad, clusterize_thresh)
644  if opt_diff2:
645  bad = np.concatenate( ([bad[0]], bad, [bad[-1]]) )
646 
647  bad_ = np.copy(bad)
648  if offset>0:
649  for i,b in enumerate(bad):
650  if b:
651  bad_[i] = True
652  for j in range(offset):
653  if (i+j+1)<len(bad): bad_[i+j+1] = True
654  if returnval:
655  return bad_, thre
656  else:
657  return bad_

660  return (y[2:]+y[:-2]-2*y[1:-1])/dx**2

662 def _clusterize_indices(indices, threshold):
663  """Fill small gap (within `threshold`) in indices.
664  引数
665  ----
666  indices: an 1-D array of Boolean
667  threshold: an interger, allowed gap between True's in indices
668 
669  戻り値
670  ------
671  list 真理値のリスト
672 
673  実行結果の例
674  ------------
675  If threshold == 1,
676  [True, False, False, True, True] => [True, False, False, True, True].
677  If threshold == 2,
678  [True, False, False, True, True] => [True, True, True, True, True].
679  """
680 
681  results = np.copy(indices)
682  prev = 0
683  first = True
684  for i, x in enumerate(results):
685  if x:
686  if (not first) and i - prev <= threshold + 1:
687  for j in range(prev, i):
688  results[j] = True
689  prev = i
690  first = False
691  return results

880 def power_spectrum_density(data, dt, ndivide=1, window=scipy.signal.hanning, overwrap_half=False):
881  """Calculate power spectrum density of data.
882  引数
883  ----
884  data : input data
885  dt : time between each data
886  ndivide : do averaging(split data into ndivide, get psd of each, and average them)
887  overwrap_half: a boolean, split data to half-overwrapped regions
888 
889  戻り値
890  ------
891  (frequencies, psd)
892  """
893  if overwrap_half:
894  step = int(len(data)/(ndivide + 1))
895  size = step*2
896  else:
897  step = int(len(data)/ndivide)
898  size = step
899  if bin(size).count("1") != 1:
900  if overwrap_half:
901  warnings.warn('warning: ((length of data)/(ndivide+1))*2 is not power of 2: %d' % size)
902  else:
903  warnings.warn('warning: (length of data)/ndivide is not power of 2: %d' % size)
904  psd = np.zeros(size)
905  T = (size - 1)*dt
906  vs = 1/dt
907  vk_ = scipy.fft.fftfreq(size, dt)
908  vk = vk_[np.where(vk_ >= 0)]
909  for i in range(ndivide):
910  d = data[i*step:i*step+size]
911  if window is None:
912  w = np.ones(size)
913  corr = 1.0
914  else:
915  w = window(size)
916  corr = np.mean(w**2)
917  psd = psd + 2*(np.abs(scipy.fft.fft(d*w)))**2/size*dt/corr
918  return vk, psd[:len(vk)]/ndivide
