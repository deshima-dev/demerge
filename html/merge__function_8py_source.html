144 def Tlos_model(dx, p0, etaf, T0, Troom, Tamb):
145  """Calibrate 'amplitude' and 'phase' to 'power'"""

148 def calibrate_to_power(pixelid, Troom, Tamb, rhdus, ddb):
149  nkid = rhdus['READOUT'].header['NKID%d' %pixelid]
150  kiddict = {}
151  for (i, j) in zip(ddb['KIDFILT'].data['kidid'], ddb['KIDFILT'].data['masterid']):
152  kiddict[i] = j
153 
154  linphase = np.transpose([rhdus['READOUT'].data['Amp, Ph, linPh %d' %i].T[2] for i in range(nkid)])

157  fshift = np.array((linphase - linyfc)/ (4.*Qr)).T
158  fshift_err = np.zeros( len(fshift) )
159  #---- Responsivity curve

161  Tsignal = []

163  masterid = kiddict[i]

171 def get_Troom(rhdus, cabin_db):
172  """Get Troom from cabin temperature database"""
173  starttime = convert_timestamp(rhdus['READOUT'].data['timestamp'])
174  margin = timedelta(seconds=CABIN_Q_MARGIN)
175  db_start = datetime.strptime(starttime[0], FORM_FITSTIME_P)
176  db_start_mod = db_start - timedelta(microseconds=db_start.microsecond) - margin
177  db_end = datetime.strptime(starttime[-1], FORM_FITSTIME_P)
178  db_end_mod = db_end - timedelta(microseconds=db_end.microsecond) + margin
179  db_ret = cabin_db.get_between(db_start_mod.isoformat(), db_end_mod.isoformat())
180  if len(db_ret)==0:
181  print('Cannot find cabin temperature data in', cabin_db.db_path, file=sys.stderr)
182  print('Instead, Troom', DEFAULT_ROOM_T, 'K will be used', file=sys.stderr)
183  Troom = DEFAULT_ROOM_T
184  else:
185  # Check whether the database covers the measurement
186  db_dts = [datetime.strptime(entry[0], FORM_FITSTIME) for entry in db_ret]
187  main_t = [entry[2] for entry in db_ret]
188  if (db_start < db_dts[0]) or (db_dts[-1] < db_end):
189  print('DB data does not cover the measurement span', file=sys.stderr)
190  print('Temperature at the edge(s) will be used to cover the range', file=sys.stderr)
191  # Expand
192  if (db_start < db_dts[0]):
193  db_dts = [db_start] + db_dts
194  main_t = [main_t[0]] + main_t
195  if (db_dts[-1] < db_end):
196  db_dts = db_dts + [db_end]
197  main_t = main_t + [main_t[-1]]
198 
199  starttime_ut = rhdus['READOUT'].data['timestamp']
200  db_uts = [timegm(dt.timetuple()) for dt in db_dts]
201 
202  int_func = scipy.interpolate.interp1d(db_uts, main_t)
203  Troom = int_func(starttime_ut) + 273.15
204  return Troom, db_ret
