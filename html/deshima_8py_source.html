167 def fitLorentzian(freq, ampl, f0, q0):
168  """
169  Fit data with lorenzian curve.
170 
171  :param freq: a 1-D array of frequency
172  :param ampl: a 1-D array of amplitude
173  :param f0: initial parameter for center frequency
174  :param q0: initial parameter for quality factor
175 
176  :return: (fc, q, d, bg)
177 
178  - **fc**: fit of center frequency
179  - **q**: fit of quality factor
180  - **d**: fit of amplitude for Lorentzian curve
181  - **bg**: fit of constant background level
182  """
183  def f(x):
184  (a, b, c, d) = x # background, amplitude, 2/FWHM, freq. center
185  y = a + b / (((freq-d)*c)**2 + 1)
186  return y - ampl
187  a = np.median(ampl)
188  b = -0.8 * a
189  c = 2.0 * q0 / f0
190  d = f0
191  x0 = np.array([a, b, c, d])
192  x1 = scipy.optimize.leastsq(f, x0)[0]
193  (a, b, c, d) = x1
194  fc = d
195  q = abs(c * d / 2.0)
196  return (fc, q, -b, a)

218 def adjust_fitrange(nfwhm, ndata, nparams, peakparams, factor=1):
219  """
220  Adjust fit range to make sure :math:`N_{free} \geq nparam`
221 
222  :param nfwhm: a float, minimum fit length in unit of FWHM
223  :param ndata: a integer, length of data
224  :param nparams: a integer, number of parameter
225  :param peakparams: a dict of peak information
226  :param factor: a integer
227 
228  :Note: *specify 2* as **factor** if you fit complex variable (as 2d-vector)
229 
230  **peakparams** are:
231 
232  - **f0ind**: a integer, index of peak center
233  - **dl**: a integer, index count to reach the left half-maximum
234  - **dr**: a integer, index count to reach the right half-maximum
235 
236  :return: (rbegin, rend)
237 
238  - **rbegin** is a integer of fit range begin
239  - **rend** is a integer of fit range end
240  """
241  f0ind = peakparams['f0ind']
242  l, c, r = f0ind-peakparams['dl'], f0ind, f0ind+peakparams['dr']
243  if (r - l + 1)*factor >= nparams:
244  rbegin, rend = int(c-nfwhm*(c-l)), int(c+nfwhm*(r-c))
245  else:
246  n = nparams/(float(r - l + 1)*factor)
247  rbegin, rend = int(f0ind-nfwhm*n*(c-l)), int(c+nfwhm*n*(r-c))
248  if rbegin < 0:
249  if rend + (-rbegin) >= ndata:
250  rend = ndata-1
251  rbegin = 0
252  else:
253  rend = rend + (-rbegin)
254  rbegin = 0
255  if rend >= ndata:
256  if rbegin - (rend - ndata) < 0:
257  rbegin = 0
258  rend = ndata-1
259  else:
260  rbegin = rbegin - (rend - ndata)
261  rend = ndata-1
262  if (rend - rbegin + 1)*factor < 11:
263  raise MKIDDataException("Fit range guess error")
264  return slice(rbegin, rend)

841  """
